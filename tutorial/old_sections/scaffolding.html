Our app is going to target document.body. Therefore, there is almost nothing going on in our one .html file. In your favorite text editor, open mithril_tutorial_1/src/index.html, and get it into game shape:

src/index.html

<!doctype html>
<html>
<head>
    <meta charset="utf-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1" />
    <title>My Application</title>
</head>
<body>
	<script src="./index.js"></script>
</body>
</html>
Next, create index.js inside the src folder:

src/index.js

var m = require("mithril")
	
var App = {
	view: function(){
		return m("", "Introducing modules!")
	}
}
	
m.mount(document.body, App)
