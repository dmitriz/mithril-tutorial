Ternaries

Let's display the full name and id of the most recently clicked teacher, and highlight their name in the list.

To achieve this, we'll use 2 ternaries: one to conditionally set a class on the current selection in the list, and another to conditionally display the details of the current selection.

src/index.js

var m = require("mithril")
	
var Teachers = [ ... ] // should still contain the 10 teachers from the last exercise
var current = null
	
function fullName(teacher) {
	return teacher.firstName + " " + teacher.lastName
}
	
var App = {
	view: function() {
		return m(".list"
			, Teachers.map(function(teacher){
				return m(".list-item"
					, {
						// here we use a ternary to set a class only on the selected object
                        class: teacher === current ? "current" : "",
						onclick: function() { current = teacher	}
					}
					, fullName(teacher))
			})
			// and here's a ternary which omits the detail until something is selected
			, current
                ? m('.detail', "Name: " + fullName(teacher) + ", ID: " + teacher.id) 
                : null
		)
	}
}
	
m.mount(document.body, App)

We'll also add some styles. Create styles.css in src:

src/styles.css

* {	box-sizing: border-box;	font-family: "Helvetica Neue", Calibri, Roboto, Arial, sans-serif; }
.list-item { padding: 8px; background-color: rgba(0, 0, 0, 0.05); border: 1px solid #222; margin: 5px 0; transition: background-color 0.3s; }
.list-item.current { background-color: rgb(56, 255, 38); }
.list-item:hover { cursor: pointer; text-decoration: underline; }
.detail { padding: 10px; background: #555555; color: white}
